Cheatsheet for SQL:

SQL Queries:

Description

Example

Select all data for table:
select * from [name of table];

 name of table: states
select * from states;

Select specific column(s) from a table:
select [name of column], [name of column]
from [name of table];

name of columns: state_name, population;
name of table: states:
select state_name, population
from states;

Select specific columns and ordered by a condition.
Order in descending/ascending order
select [name of column], [name of column]
from [name of table]
order by [condition] [DESC (for descending order)/ASC (default, for ascending order);

name of columns: state_name, population;
name of table: states;
condition: ordered by population
from highest to lowest;
select state_name, population
from states
order by population DESC;

Select a column from a table with specified condition in the column:
select [name of column]
from [name of table]
where [specified condition of column];

name of column: state_name;
name of table: states
specified condition of column: in region 7:
select state_name
from states
where region_id=7;

Select specified columns from a table with
specific ordered condition for the column
select [name of column]
from [name of table]
where [condition for column]
order by [name of column];

name of columns: state_name,
population_density;
name of table: states
condition of column: with a population
density over 50:
ordered from least to most dense:
select state_name, population_density
from states
where population_density>50
order by population_density;

Select column from a table with range values for a specified condition
select [name of column]
from [name of table]
where [specified condition and beginning of range value]
and [specified condition and end of range value];

name of column:state_name;
name of table: states;
condition and range values:
population between 1 and 1.5 million people:
select state_name
from states
where population<1000000
and population>1500000;

Select columns from a table and
order by specific condition:
select [name of columns]
from [name of table]
order by [specific condition];

name of columns: state_name,region_id;
name of table: states
condition: region in ascending order:
select state_name, region_id
from states
order by region_id;

Select name of column
in name of table with specified
text in the table:
select [name of column]
from [name of table]
where [name of column] like [specfied text];

name of column:region_name
name of table: regions
specified text: "Central"
select region_name
from regions
where region_name like '%Central%';

Select name of columns from
two tables in order by correlating
name of columns:
select [name of table1].[name of column], [name of table2].[name of column]
from [name of table 1]
inner join [name of table 2]
on [table1].[column name]=[table2][column name];

name of column/table 1: region_name;
name of column/table2: state_name;
name of table1: regions
name of table2: states
in ascending order by region_id:
select regions.region_name, states.state_name
from regions
inner join states
on regions.id=states.region_id;
